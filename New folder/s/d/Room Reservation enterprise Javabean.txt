package ejb;
import javax.ejb.Stateless;
@Stateless
public class RoomBean
{
 int eco_amt=1000;
 int delx_amt=3000;
 int exe_amt=5000;
 private int exec_amt;

 public int Cal(String room_type,int day_num)
 {
     int result=0;
 if(room_type.equals("economy"))
 {
     result=(eco_amt*day_num);
  }
 
else if(room_type.equals("deluxe"))
 {
     result=(delx_amt*day_num);
 }
 else if(room_type.equals("executive"))
 {
     result=(exe_amt*day_num);
 }
     return result;
 }
 }
