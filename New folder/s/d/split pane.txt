import java.awt.*;
import java.awt.event.*;
import javax.swing.*;
import javax.swing.event.*;

public class practical3 extends JFrame implements ListSelectionListener
{
JList plist;
JLabel lblimg;
String planets[]={"Mercury","Neptune","Pluto"};
public practical3()
{
setTitle("SplitPane Program");
setSize(400,400);
plist=new JList(planets);
plist.addListSelectionListener(this);
lblimg=new JLabel();
JSplitPane ip=new JSplitPane(JSplitPane.HORIZONTAL_SPLIT,plist,lblimg);
ip.setContinuousLayout(true);
ip.setOneTouchExpandable(true);
add(ip,BorderLayout.CENTER);
}

public void valueChanged(ListSelectionEvent e)
{
String str=plist.getSelectedValue().toString();
ImageIcon img=new ImageIcon(str +".jpg");
lblimg.setIcon(img);
}
public static void main(String args[])
{
practical3 p=new practical3();
p.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
p.setVisible(true);
}
}
