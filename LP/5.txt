#include<iostream>

int main()
{
int n,i,first=0,second=1,next;
cout<<"enter number &terms of fibonacci series:";
cin>>n;
cout<<"first"<<n<<"terms of fibonacci series";
for(i=0;i<n;i++)
{
if(i<=1)
next=1;
else
{
next=first+second;
first=second;
second=next;
}
cout<<next<<endl;
}
return 0;
}

#include<iostream>
using namespace std;
int main()
{
int num;
cout<<"enter a no:";
cin>>num;
int i=2;
while(i<=num-1)
{
if(num%i==0)
{
cout<<"\n"<<num<<"is not prime";
break;
}
i++;
}
if(i==num)
cout<<"\n"<<num<<"is a prime\n";
return 0;
}
