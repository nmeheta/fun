import java.awt.*;
importjava.awt.event.*;
importjavax.swing.*;
importjavax.swing.event.*;
public class Pract1 extends JFrame implements ListSelectionListener
{
JList jlst;
JLabel  l1,l2;
JScrollPane jsp;
Int idx;
String items[]={"Notebook","Eraser","Sharpner","Ball Pen"};
public Pract1()
{
getContentPane().setLayout(new BorderLayout());
jlst=new JList(items);
jlst.addListSelectionListener(this);
jlst.setSelectionMode(ListSelectionModel.SINGLE_SELECTION);
jsp=new JScrollPane(jlst);
l1=new JLabel("Select Product");
l2=new JLabel("Select Price");
getContentPane().add(l1,BorderLayout.NORTH);
getContentPane().add(l2,BorderLayout.SOUTH);
getContentPane().add(jsp,BorderLayout.CENTER);
}
public void valueChanged(ListSelectionEvent e)
{
idx=jlst.getSelectedIndex();
if(idx!=-1)
{
l1.setText("Current Selection"+items[idx]);
show(idx);
}
else
{
l1.setText("Select any product");
}}
public void show(int i)
{
if(i==0)
{
l2.setText("Price of"+items[i]+"is:20 Rs");
}
else if(i==1)
{
l2.setText("Price of"+items[i]+"is:5 Rs");
}
else if(i==2)
{
l2.setText("Price of"+items[i]+"is:10 Rs");
}
else
{
l2.setText("Price of"+items[i]+"is:7 Rs");
}}
public static void main(String args[])
{
Pract1 p1=new Pract1();
p1.setTitle("Stationary");
p1.setSize(300,200);
p1.setVisible(true);
p1.setDefaultCloseOperation(EXIT_ON_CLOSE);
} }
